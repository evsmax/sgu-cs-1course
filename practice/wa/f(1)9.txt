#include <iostream>
#include <fstream>
#include <string>


int main()
{
    std::size_t k1, k2;
    std::cin >> k1 >> k2;


    std::string line;


    std::ifstream fin;
    fin.open("D:\\file.txt", std::ios::in);
    while (!fin.eof())
    {    
        std::getline(fin, line);
        std::cout << std::string(line.begin() + k1, line.begin() + k2);
    }


    return 0;
}
